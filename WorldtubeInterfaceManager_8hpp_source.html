21 /*!
22  * \brief Abstract base class for storage and retrieval of generalized harmonic
23  * quantities communicated from a Cauchy simulation to the Cce system.
24  *
25  * \details The functions that are required to be overriden in the derived
26  * classes are:
27  * - `GhWorldtubeInterfaceManager::get_clone()`: should return a
28  * `std::unique_ptr<GhWorldtubeInterfaceManager>` with cloned state.
29  * - `GhWorldtubeInterfaceManager::insert_gh_data()`: should store the portions
30  * of the provided generalized harmonic data that are required to provide useful
31  * boundary values for the CCE evolution at requested timesteps.
32  * - `GhWorldtubeInterfaceManager::request_gh_data()`: should register requests
33  * from the CCE evolution for boundary data.
34  * - `GhWorldtubeInterfaceManager::retrieve_and_remove_first_ready_gh_data()`:
35  * should return a `boost::optional<std::tuple<TimeStepId, tnsr::aa<DataVector,
36  * 3>, tnsr::iaa<DataVector, 3>, tnsr::aa<DataVector, 3>>>` containing the
37  * boundary data associated with the oldest requested timestep if enough data
38  * has been supplied via `insert_gh_data()` to determine the boundary data.
39  * Otherwise, return a `boost::none` to indicate that the CCE system must
40  * continue waiting for generalized harmonic input.
41  * - `GhWorldtubeInterfaceManager::number_of_pending_requests()`: should return
42  * the number of requests that have been registered to the class that do not yet
43  * been retrieved via `retrieve_and_remove_first_ready_gh_data()`.
44  * - `GhWorldtubeInterfaceManager::number_of_gh_times()`: should return the
45  * number of time steps sent to `insert_gh_data()` that have not yet been
46  * retrieved via `retrieve_and_remove_first_ready_gh_data()`.
47  */
48 class GhWorldtubeInterfaceManager : public PUP::able {
49  public:
50  using creatable_classes = tmpl::list<GhLockstepInterfaceManager>;
51 
52  WRAPPED_PUPable_abstract(GhWorldtubeInterfaceManager); // NOLINT
53 
54  virtual std::unique_ptr<GhWorldtubeInterfaceManager> get_clone() const
55  noexcept = 0;
56 
57  virtual void insert_gh_data(TimeStepId time_id,
58  tnsr::aa<DataVector, 3> spacetime_metric,
59  tnsr::iaa<DataVector, 3> phi,
60  tnsr::aa<DataVector, 3> pi,
61  tnsr::aa<DataVector, 3> dt_spacetime_metric,
62  tnsr::iaa<DataVector, 3> dt_phi,
63  tnsr::aa<DataVector, 3> dt_pi) noexcept = 0;
64 
65  virtual void request_gh_data(const TimeStepId&) noexcept = 0;
66 
67  virtual auto retrieve_and_remove_first_ready_gh_data() noexcept
68  -> boost::optional<
69  std::tuple<TimeStepId, tnsr::aa<DataVector, 3>,
70  tnsr::iaa<DataVector, 3>, tnsr::aa<DataVector, 3>>> = 0;
71 
72  virtual size_t number_of_pending_requests() const noexcept = 0;
73 
74  virtual size_t number_of_gh_times() const noexcept = 0;
75 };

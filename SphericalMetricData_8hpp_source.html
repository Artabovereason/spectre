25 /*!
26  * \brief Abstract base class for analytic worldtube data most easily derived in
27  * spherical coordinate form.
28  *
29  * \details This class provides the functions required by the `WorldtubeData`
30  * interface that convert from a spherical coordinate spacetime metric to
31  * Cartesian coordinates. Derived classes of `SphericalMetricData` need not
32  * implement the `variables_impl`s for the Cartesian quantities. Instead, the
33  * derived classes must override the protected functions:
34  * - `SphericalMetricData::spherical_metric()`
35  * - `SphericalMetricData::dr_spherical_metric()`
36  * - `SphericalMetricData::dt_spherical_metric()`
37  * Derived classes are still responsible for overriding
38  * `WorldtubeData::get_clone()`, `WorldtubeData::variables_impl()` for tag
39  * `Cce::Tags::News`, and `WorldtubeData::prepare_solution()`.
40  */
41 struct SphericalMetricData : public WorldtubeData {
42  /*!
43  * Computes and returns by pointer the Jacobian
44  * \f$\partial x_{\mathrm{spherical}}^j / \partial
45  * x_{\mathrm{Cartesian}}^i\f$
46  */
47  void inverse_jacobian(gsl::not_null<CartesianiSphericalJ*> inverse_jacobian,
48  size_t l_max) const noexcept;
49 
50  /*!
51  * Computes and returns by pointer the first radial derivative of the
52  * Jacobian: \f$\partial_r (\partial x_{\mathrm{spherical}}^j / \partial
53  * x_{\mathrm{Cartesian}}^i)\f$
54  */
55  void dr_inverse_jacobian(
56  gsl::not_null<CartesianiSphericalJ*> dr_inverse_jacobian,
57  size_t l_max) const noexcept;
58 
59  protected:
60  using WorldtubeData::variables_impl;
61 
62  /*!
63  * \brief Computes the Cartesian spacetime metric from the spherical solution
64  * provided by the derived classes.
65  *
66  * \details The derived classes provide spherical metric data via the virtual
67  * function `SphericalMetricData::spherical_metric()` at a resolution
68  * determined by member variable `l_max_`. This function performs the
69  * coordinate transformation using the Jacobian computed from
70  * `SphericalMetricData::inverse_jacobian()`.
71  */
72  void variables_impl(gsl::not_null<tnsr::aa<DataVector, 3>*> spacetime_metric,
73  size_t l_max, double time,
74  tmpl::type_<gr::Tags::SpacetimeMetric<
75  3, ::Frame::Inertial, DataVector>> /*meta*/) const
76  noexcept override;
77 
78  /*!
79  * \brief Computes the time derivative of the Cartesian spacetime metric from
80  * the spherical solution provided by the derived classes.
81  *
82  * \details The derived classes provide the time derivative of the spherical
83  * metric data via the virtual function
84  * `SphericalMetricData::dt_spherical_metric()` at a resolution determined by
85  * member variable `l_max_`. This function performs the coordinate
86  * transformation using the Jacobian computed from
87  * `SphericalMetricData::inverse_jacobian()`.
88  */
89  void variables_impl(
90  gsl::not_null<tnsr::aa<DataVector, 3>*> dt_spacetime_metric, size_t l_max,
91  double time,
92  tmpl::type_<::Tags::dt<gr::Tags::SpacetimeMetric<
93  3, ::Frame::Inertial, DataVector>>> /*meta*/) const noexcept override;
94 
95  /*!
96  * \brief Computes the spatial derivatives of the Cartesian spacetime metric
97  * from the spherical solution provided by the derived classes.
98  *
99  * \details The derived classes provide the radial derivative of the spherical
100  * metric data via the virtual function
101  * `SphericalMetricData::dr_spherical_metric()` at a resolution determined by
102  * member variable `l_max_`. This function performs the additional angular
103  * derivatives necessary to assemble the full spatial derivative and performs
104  * the coordinate transformation to Cartesian coordinates via the Jacobians
105  * computed in `SphericalMetricData::inverse_jacobian()` and
106  * `SphericalMetricData::inverse_jacobian()`.
107  */
108  void variables_impl(
109  gsl::not_null<tnsr::iaa<DataVector, 3>*> d_spacetime_metric, size_t l_max,
110  double time,
111  tmpl::type_<
112  GeneralizedHarmonic::Tags::Phi<3, ::Frame::Inertial>> /*meta*/) const
113  noexcept override;
114 
115  /// Must be overriden in the derived class; should compute the spacetime
116  /// metric of the analytic solution in spherical coordinates.
117  virtual void spherical_metric(
118  gsl::not_null<
119  tnsr::aa<DataVector, 3, ::Frame::Spherical<::Frame::Inertial>>*>
120  spherical_metric,
121  double time) const noexcept = 0;
122 
123  /// Must be overriden in the derived class; should compute the first radial
124  /// derivative of the spacetime metric of the analytic solution in spherical
125  /// coordinates.
126  virtual void dr_spherical_metric(
127  gsl::not_null<
128  tnsr::aa<DataVector, 3, ::Frame::Spherical<::Frame::Inertial>>*>
129  dr_spherical_metric,
130  double time) const noexcept = 0;
131 
132  /// Must be overriden in the derived class; should compute the first time
133  /// derivative of the spacetime metric of the analytic solution in spherical
134  /// coordinates.
135  virtual void dt_spherical_metric(
136  gsl::not_null<
137  tnsr::aa<DataVector, 3, ::Frame::Spherical<::Frame::Inertial>>*>
138  dt_spherical_metric,
139  double time) const noexcept = 0;
140 };
